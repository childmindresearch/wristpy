 54class GgirCalibration(AbstractCalibrator):
 55    """Implements the GGIR calibration on accelerometer data.
 56
 57    This class implements methods for autocalibrating accelerometer data using either
 58    entire dataset or subsets, as determined by the settings. Depending on the settings
 59    a scale and offset value is determined and applied to the data which, if
 60    successful, is returned to the user.
 61
 62
 63    Attributes:
 64        chunked: If true will preform calibration on subsets of data if false will
 65            calibrate on entire dataset.
 66        min_acceleration: Minimum acceleration for sphere criteria, in g-force.
 67        min_calibration_hours: Minimum hours of data required for calibration.
 68        no_motion_threshold: Minimum standard deviation for no-motion detection.
 69        max_iterations: Maximum number of iterations for optimization.
 70        error_tolerance: Tolerance for optimization convergence.
 71        max_calibration_error: Threshold for the maximum allowable calibration error.
 72
 73    """
 74
 75    def __init__(
 76        self,
 77        chunked: bool = False,
 78        min_acceleration: float = 0.3,
 79        min_calibration_hours: int = 72,
 80        no_motion_threshold: float = 0.013,
 81        max_iterations: int = 1000,
 82        error_tolerance: float = 1e-10,
 83        max_calibration_error: float = 0.01,
 84    ) -> None:
 85        """Initializes class.
 86
 87        Attributes:
 88            chunked: Determines if either entire dataset will be used, or if
 89                calibration will be attempted with a subset of the data. Set to false by
 90                default, when true will initiate _chunked_calibration from the run()
 91                method instead of standard _calibration.
 92            min_acceleration: The value on either side of 0g for each axis.
 93                Determines if sphere is sufficiently populated to obtain meaningful
 94                calibration result. Default is 0.3g.
 95            min_calibration_hours: The minimum amount of data in hours needed to
 96                preform calibration. Default is 72. If chunked calibration is selected
 97                this will be the size of the initial subset taken for calibration. If
 98                error has not been reduced below the max_calibration_error value, an
 99                additional 12 hours will be taken until all data is used or calibration
100                is successful.
101            no_motion_threshold: The standard deviation critieria used to select
102                portions of the data with no movement. Rolling windows with standard
103                deviations below this value will be determined as "still". This value is
104                likely to be different between devices. Default is 0.013g. This value
105                was determined just above the empirically derived  baseline standard
106                deviation due to noise (0.010g)(van Hees et al. 2014). If measuring the
107                noise in a bench-top test, this threshold should be about `1.2 * noise`.
108            max_iterations: The maximum amount of iterations for the closest_point_fit
109                method. Default is 1000, generally should be left at this value.
110            error_tolerance: Tolerated level of error, when the
111                closest_point_fit method arrives at this value or better, the process
112                ends. Default is 1e-10. Generally should be left at this value.
113            max_calibration_error: Maximum acceptable calibration error. If calibration
114                cannot reach this threshold it will error.
115
116        Returns:
117            None
118        """
119        self.chunked = chunked
120        self.min_acceleration = min_acceleration
121        self.min_calibration_hours = min_calibration_hours
122        self.no_motion_threshold = no_motion_threshold
123        self.max_iterations = max_iterations
124        self.error_tolerance = error_tolerance
125        self.max_calibration_error = max_calibration_error
126
127    def run_calibration(self, acceleration: models.Measurement) -> models.Measurement:
128        """Runs calibration on acceleration data based on settings.
129
130        If the chunked arguement is set to true, it will run calibration on an initial
131        chunk of the data the size of which is set to min_calibration_hours. If it fails
132        to calibrate based on this subset, it will add 12 hour chunks to the subset
133        until calibration succeeds, or fails on the entire dataset. Otherwise if chunked
134        is false (default) calibration will be conducted on the whole of the dataset.
135        Calibration is successful when scale and offset values, that sufficiently
136        minimize error when applied to the data, are found. The scale and offset values
137        are then applied to every data point in the dataset which is then returned as
138        calibrated models.Measurement object.
139
140        Args:
141            acceleration: the accelerometer data containing x,y,z axis
142                data and time stamps.
143
144        Returns:
145            A Measurement object that contains the calibrated acceleration data.
146
147        Raises:
148            ValueError: If the acceleration data does not meet the specified minimum
149                hours of data as given by min_calibration_hours.
150            SphereCriteriaError: If the sphere is not sufficiently populated, i.e. every
151                axis does not have at least 1 value both above and below  the + and
152                - value of min_acceleraiton.
153            CalibrationError: If the calibration process fails to get below the
154                `max_calibration_error` threshold.
155        """
156        logger.debug("Starting calibration.")
157        data_range = cast(datetime, acceleration.time.max()) - cast(
158            datetime, acceleration.time.min()
159        )
160        total_hours = math.floor(data_range.total_seconds() / 3600)
161
162        if total_hours < self.min_calibration_hours:
163            raise exceptions.CalibrationError(
164                (
165                    f"Calibration requires {self.min_calibration_hours} hours"
166                    f"but only {total_hours} hours of data were given."
167                ),
168            )
169
170        if self.chunked:
171            linear_transformation = self._chunked_calibration(acceleration=acceleration)
172        else:
173            linear_transformation = self._calibrate(acceleration=acceleration)
174
175        calibrated_acceleration = (
176            acceleration.measurements * linear_transformation.scale
177        ) + linear_transformation.offset
178
179        return models.Measurement(
180            measurements=calibrated_acceleration, time=acceleration.time
181        )

183    def _chunked_calibration(
184        self, acceleration: models.Measurement
185    ) -> LinearTransformation:
186        """Chunks the data into subsets, to calibrate on smaller sections of data.
187
188        The first chunk is determined by the min_calibration_hours. Should calibration
189        fail on that subset of data, additional chunks of data are added to attempt
190        calibration on. This continues until all data is tried, or calibration succeeds.
191        Chunks are added in increments of 12 hours.
192
193        Args:
194            acceleration: the accelerometer data containing x,y,z axis
195                data and time stamps.
196
197        Returns:
198            A LinearTransformation object with scale and offset attributes to be applied
199            to acceleration data for calibration.
200
201        Raises:
202            CalibrationError: If all possible chunks have been used and the calibration
203                process fails to get below the `max_calibration_error` threshold.
204        """
205        logger.debug("Running chunked calibration.")
206        for chunk in self._get_chunk(acceleration):
207            try:
208                return self._calibrate(chunk)
209            except (
210                exceptions.SphereCriteriaError,
211                exceptions.CalibrationError,
212                exceptions.NoMotionError,
213            ):
214                pass
215        raise exceptions.CalibrationError(
216            "After all chunks of data used calibration has failed."
217        )

219    def _get_chunk(
220        self, acceleration: models.Measurement
221    ) -> Generator[models.Measurement, None, None]:
222        """Takes a subset of acceleration data to be used for calibration.
223
224        Args:
225            acceleration: the accelerometer data containing x,y,z axis
226                data and time stamps.
227
228        Returns:
229            The minimum hours of accelerometer data + additional 12 hour chunks of data
230            everytime the generator function is called.
231
232        """
233        logger.debug("Getting chunk.")
234        sampling_rate = GgirCalibration._get_sampling_rate(timestamps=acceleration.time)
235        min_samples = int(self.min_calibration_hours * 3600 * sampling_rate)
236        chunk_size = int(12 * 3600 * sampling_rate)
237        total_samples = len(acceleration.measurements)
238        if min_samples == total_samples:
239            sample_indices = [total_samples]
240        else:
241            sample_indices = list(range(min_samples, total_samples, chunk_size))
242            if sample_indices[-1] != total_samples:
243                sample_indices += [total_samples]
244
245        for idx in sample_indices:
246            yield models.Measurement(
247                measurements=acceleration.measurements[:idx, :],
248                time=acceleration.time[:idx],
249            )

251    def _calibrate(self, acceleration: models.Measurement) -> LinearTransformation:
252        """Calibrates data and returns scale and offset values.
253
254        The acceleration data is processed by the _extract_no_motion function, which
255        returns the portions of the data where the subject was still. This data ideally
256        should all have points with a norm of 1. If we take a unit sphere, all the
257        points in no_motion_data should be found along it's surface. As this is not the
258        case we calibrate by finding offset and scale values that will most closely make
259        the whole of our data lie along this unit sphere. This is done by the
260        _closest_point_fit function. We then transform the data by applying the scale
261        and offset values we have found, and if the error has been sufficiently reduced,
262        these values are returned. If not calibration fails and raises a calibration
263        error.
264
265        Args:
266            acceleration: the accelerometer data containing x,y,z axis
267                data and time stamps.
268
269        Returns:
270            A LinearTransformation object with scale and offset attributes to be applied
271            to acceleration data for calibration.
272
273        Raises:
274            CalibrationError: If the calibration process fails to converge or the final
275                error exceeds the `max_calibration_error` threshold.
276
277        References:
278            van Hees VT, Fang Z, Langford J, et al. Autocalibration of accelerometer
279            data for free-living physical activity assessment using local gravity
280            and temperature: an evaluation on four continents. J Appl Physiol (1985)
281            2014 Oct 1;117(7):738-44. doi: 10.1152/japplphysiol.00421.2014.
282        """
283        logger.debug("Attempting to calibrate...")
284        no_motion_data = _extract_no_motion(acceleration=acceleration)
285
286        linear_transformation = self._closest_point_fit(no_motion_data=no_motion_data)
287
288        no_motion_calibrated = (
289            no_motion_data * linear_transformation.scale
290        ) + linear_transformation.offset
291
292        cal_error_initial = np.round(
293            np.mean(abs(np.linalg.norm(no_motion_data, axis=1) - 1)), decimals=5
294        )
295        cal_error_end = np.around(
296            np.mean(abs(np.linalg.norm(no_motion_calibrated, axis=1) - 1)), decimals=5
297        )
298
299        if (cal_error_end > cal_error_initial) or (
300            cal_error_end >= self.max_calibration_error
301        ):
302            raise exceptions.CalibrationError(
303                "Calibration error could not be sufficiently minimized. "
304                f"Initial Error: {cal_error_initial}, Final Error: {cal_error_end}, "
305                f"Error threshold: {self.max_calibration_error}"
306            )
307        logger.debug(
308            "Calibration successful. Scale: %s, Offset: %s",
309            linear_transformation.scale,
310            linear_transformation.offset,
311        )
312        return linear_transformation

314    def _closest_point_fit(self, no_motion_data: np.ndarray) -> LinearTransformation:
315        """Applies closest point fit to no motion data to calibrated accelerometer data.
316
317        This method implements an iterative algorithm that finds the optimal scale and
318        offset for calibrating accelerometer data. The ndarray it processes contains the
319        periods of "no motion" identified by the _extract_no_motion method. This data is
320        then fit to an idealized unit sphere which represents the accelerometer data
321        under noiseless conditions. The scale and offset derived from this process are
322        used to calibrate our data along each axis.
323
324        Args:
325            no_motion_data: The acceleration data during periods of no
326                motion, in order to determine scale and offset.
327
328        Returns:
329            A LinearTransformation object with scale and offset attributes to be applied
330            to acceleration data for calibration.
331
332        Raises:
333            SphereCriteriaError: If the sphere is not sufficiently populated, i.e. every
334                axis does not have at least 1 value both above and below  the + and
335                - value of min_acceleraiton.
336            ZeroScaleError: Numerical instability inherent to the algorithm may cause
337                values of the scale vector to tend toward zero under certain conditions.
338                This will cause calibration to fail.
339
340        References:
341            van Hees VT, Fang Z, Langford J, et al. Autocalibration of accelerometer
342            data for free-living physical activity assessment using local gravity
343            and temperature: an evaluation on four continents. J Appl Physiol (1985)
344            2014 Oct 1;117(7):738-44. doi: 10.1152/japplphysiol.00421.2014.
345        """
346        logger.debug("Beginning closest point fit.")
347        sphere_criteria_check = np.all(
348            (no_motion_data.min(axis=0) < -self.min_acceleration)
349            & (no_motion_data.max(axis=0) > self.min_acceleration)
350        )
351        if not sphere_criteria_check:
352            raise exceptions.SphereCriteriaError(
353                "Did not meet criteria to sufficiently populate sphere"
354            )
355
356        weights = np.ones(no_motion_data.shape[0]) * 100
357        previous_residual = np.inf
358
359        linear_regression_model = linear_model.LinearRegression()
360
361        offset = np.zeros(3)
362        scale = np.ones(3)
363        for _ in range(self.max_iterations):
364            current = (no_motion_data * scale) + offset
365            closest_point = current / np.linalg.norm(current, axis=1, keepdims=True)
366            linear_regression_model.fit(current, closest_point, sample_weight=weights)
367            offset_change = linear_regression_model.intercept_
368            scale_change = np.diag(linear_regression_model.coef_)
369
370            if np.all((scale * scale_change) < 1e-8):
371                raise exceptions.CalibrationError(
372                    """Calibration has failed as a result of zero scale values."""
373                )
374
375            scale *= scale_change
376            offset += offset_change / scale
377
378            residual = 0.03 * sklearn_metrics.mean_squared_error(
379                current, closest_point, sample_weight=weights
380            )
381
382            weights = np.minimum(
383                1 / np.linalg.norm(current - closest_point, axis=1), 100
384            )
385
386            logger.debug("Scale: %s, Offset: %s, Residual: %s", scale, offset, residual)
387            if abs(residual - previous_residual) < self.error_tolerance:
388                logger.debug("Change in residual below error tolerance, ending loop.}")
389                break
390
391            previous_residual = residual
392
393        return LinearTransformation(scale=scale, offset=offset)

415class ConstrainedMinimizationCalibration(AbstractCalibrator):
416    """Calibrates accelerometer data using the default wristpy method.
417
418    This is a modification of the method proposed by Van Hees et al. (2014), in which we
419    use all available data, make use of the scipy optimize minimize function to solve
420    for the scaling and offset parameters of a LinearTransformation that will be applied
421    to the data. The minimization function looks to minimize the error between the
422    no_motion_data and the unit sphere.
423
424    Attributes:
425        acceleration: The accelerometer data that we want to calibrate.
426        no_motion_threshold: Minimum standard deviation for no-motion detection.
427    """
428
429    def __init__(
430        self,
431        no_motion_threshold: float = 0.013,
432        max_iterations: int = 1000,
433        max_calibration_error: float = 0.01,
434    ) -> None:
435        """Initializes class.
436
437        Args:
438            no_motion_threshold: The standard deviation critieria used to find
439                periods of no motion. Default is 0.013g.
440            max_iterations: The maximum amount of iterations for the
441                closest_point_fit function.
442            max_calibration_error: Maximum acceptable calibration error. If calibration
443                cannot reach this threshold it will error.
444
445        Returns:
446            None
447        """
448        self.no_motion_threshold = no_motion_threshold
449        self.max_iterations = max_iterations
450        self.max_calibration_error = max_calibration_error
451
452    def run_calibration(self, acceleration: models.Measurement) -> models.Measurement:
453        """Runs calibration on acceleration data.
454
455        Args:
456            acceleration: the accelerometer data containing x,y,z axis
457                data and time stamps.
458
459        Returns:
460            A Measurement object that contains the calibrated acceleration data.
461
462        Raises:
463            CalibrationError: If the calibration process fails to converge or the final
464                error exceeds the `max_calibration_error` threshold.
465        """
466        logger.debug("Starting calibration.")
467        no_motion_data = _extract_no_motion(
468            acceleration=acceleration,
469            no_motion_threshold=self.no_motion_threshold,
470        )
471        linear_transformation = self._closest_point_fit(no_motion_data=no_motion_data)
472
473        logger.debug(
474            "Calibration successful. Scale: %s, Offset: %s. ",
475            linear_transformation.scale,
476            linear_transformation.offset,
477        )
478
479        calibrated_acceleration = (
480            acceleration.measurements * linear_transformation.scale
481        ) + linear_transformation.offset
482
483        return models.Measurement(
484            measurements=calibrated_acceleration, time=acceleration.time
485        )

487    def _closest_point_fit(self, no_motion_data: np.ndarray) -> LinearTransformation:
488        """Find linear transformation parameters that minimzes distance to unit sphere.
489
490        This function implements the scipy optimize.minimze function to find the
491        optimial scale and offset parameters that will minimze the error function,
492        which is defined as the distance of the no_motion acceleration data
493        from the unit sphere (where the unit sphere represents the ideal data points
494        under 1g acceleartion due to Earth's gravity). The initial guess for the
495        scale and offset are chosen as 1/0, we provide constrained bounds for the
496        scale and offset parameters to avoid the the case where the scale can be
497        set to 0 and offset to 1/sqrt(3) (exact unit sphere).
498
499        Args:
500            no_motion_data: The acceleration data during periods of no motion.
501
502        Returns:
503            A LinearTransformation object with scale and offset parameters to be applied
504            to acceleration data for calibration.
505
506        Raises:
507            CalibrationError: If the optimization process fails to converge or if the
508                calibration error is not sufficiently minimized.
509        """
510        start_scale = np.ones(3)
511        start_offset = np.zeros(3)
512
513        def get_distance_to_unit_sphere(params: list[float]) -> float:
514            scale = params[:3]
515            offset = params[3:]
516            distances = np.linalg.norm((no_motion_data * scale) + offset, axis=1) - 1
517            return np.sum(distances**2)
518
519        initial_guess = np.concatenate([start_scale, start_offset])
520        data_range = no_motion_data.max() - no_motion_data.min()
521        scale_bounds = [(0.1, None)] * 3
522        offset_bounds = [(-0.5 * data_range, 0.5 * data_range)] * 3
523        bounds = scale_bounds + offset_bounds
524        result = optimize.minimize(
525            get_distance_to_unit_sphere,
526            initial_guess,
527            options={"maxiter": self.max_iterations},
528            bounds=bounds,
529        )
530
531        if result.success:
532            optimal_scale = result.x[:3]
533            optimal_offset = result.x[3:]
534            cal_error_end = np.sqrt(result.fun / len(no_motion_data))
535        else:
536            raise exceptions.CalibrationError("Optimization failed.")
537
538        if cal_error_end >= self.max_calibration_error:
539            cal_error_initial = np.mean(
540                (np.linalg.norm(no_motion_data, axis=1) - 1) ** 2
541            )
542            logger.debug(
543                "Calibration error could not be sufficiently minimized."
544                "Initial Error: %s,  Final Error: %s, Error threshold: %s.",
545                cal_error_initial,
546                cal_error_end,
547                self.max_calibration_error,
548            )
549            raise exceptions.CalibrationError(
550                "Calibration error could not be sufficiently minimized."
551            )
552
553        return LinearTransformation(scale=optimal_scale, offset=optimal_offset)

556def _extract_no_motion(
557    acceleration: models.Measurement, no_motion_threshold: float = 0.013
558) -> np.ndarray:
559    """Identifies areas of stillness using standard deviation and mean.
560
561    The function first takes a moving standard deviation and a moving mean of the
562    acceleration data in 10 second epochs. These ndarrays are then used to identify
563    the portions of the data that have a standard deviation below
564    no_motion_threshold and a mean value < 2. These epochs are determined to be
565    the periods where the accelerometer was influenced by no motion beyond the
566    force of gravity. If periods of no motion are found, the ndarray is returned,
567    to be fit to the idealized unit sphere for the purposes of calibration.
568
569    Args:
570        acceleration: the accelerometer data containing x,y,z axis
571            data and time stamps.
572        no_motion_threshold: Threshold for the standard deviaton of acceleration data
573            used to find periods of no motion.
574
575    Returns:
576        an ndarray containing the accelerometer data determined to have no motion.
577
578    Raises:
579        NoMotionError: If no portions of data meet no motion criteria as defined by
580            no_motion_check.
581
582    References:
583        van Hees VT, Fang Z, Langford J, et al. Autocalibration of accelerometer
584        data for free-living physical activity assessment using local gravity
585        and temperature: an evaluation on four continents. J Appl Physiol (1985)
586        2014 Oct 1;117(7):738-44. doi: 10.1152/japplphysiol.00421.2014.
587    """
588    logger.debug("Extracting no motion.")
589    moving_sd = computations.moving_std(acceleration, 10)
590    moving_mean = computations.moving_mean(acceleration, 10)
591    no_motion_check = np.all(
592        moving_sd.measurements < no_motion_threshold, axis=1
593    ) & np.all(np.abs(moving_mean.measurements) < 2, axis=1)
594
595    no_motion_data = moving_mean.measurements[no_motion_check]
596
597    if not np.any(no_motion_data):
598        raise exceptions.NoMotionError(
599            "Zero non-motion epochs found. Data did not meet criteria."
600        )
601
602    return no_motion_data

311            linear_transformation.offset,

315    def _closest_point_fit(self, no_motion_data: np.ndarray) -> LinearTransformation:
316        """Applies closest point fit to no motion data to calibrated accelerometer data.
317
318        This method implements an iterative algorithm that finds the optimal scale and
319        offset for calibrating accelerometer data. The ndarray it processes contains the
320        periods of "no motion" identified by the _extract_no_motion method. This data is
321        then fit to an idealized unit sphere which represents the accelerometer data
322        under noiseless conditions. The scale and offset derived from this process are
323        used to calibrate our data along each axis.
324
325        Args:
326            no_motion_data: The acceleration data during periods of no
327                motion, in order to determine scale and offset.
328
329        Returns:
330            A LinearTransformation object with scale and offset attributes to be applied
331            to acceleration data for calibration.
332
333        Raises:
334            SphereCriteriaError: If the sphere is not sufficiently populated, i.e. every
335                axis does not have at least 1 value both above and below  the + and
336                - value of min_acceleraiton.
337            ZeroScaleError: Numerical instability inherent to the algorithm may cause
338                values of the scale vector to tend toward zero under certain conditions.
339                This will cause calibration to fail.
340
341        References:
342            van Hees VT, Fang Z, Langford J, et al. Autocalibration of accelerometer
343            data for free-living physical activity assessment using local gravity
344            and temperature: an evaluation on four continents. J Appl Physiol (1985)
345            2014 Oct 1;117(7):738-44. doi: 10.1152/japplphysiol.00421.2014.
346        """
347        logger.debug("Beginning closest point fit.")
348        sphere_criteria_check = np.all(
349            (no_motion_data.min(axis=0) < -self.min_acceleration)
350            & (no_motion_data.max(axis=0) > self.min_acceleration)
351        )
352        if not sphere_criteria_check:
353            raise exceptions.SphereCriteriaError(
354                "Did not meet criteria to sufficiently populate sphere"
355            )
356
357        weights = np.ones(no_motion_data.shape[0]) * 100
358        previous_residual = np.inf
359
360        linear_regression_model = linear_model.LinearRegression()
361
362        offset = np.zeros(3)
363        scale = np.ones(3)
364        for _ in range(self.max_iterations):
365            current = (no_motion_data * scale) + offset
366            closest_point = current / np.linalg.norm(current, axis=1, keepdims=True)
367            linear_regression_model.fit(current, closest_point, sample_weight=weights)
368            offset_change = linear_regression_model.intercept_
369            scale_change = np.diag(linear_regression_model.coef_)
370
371            if np.all((scale * scale_change) < 1e-8):
372                raise exceptions.CalibrationError(
373                    """Calibration has failed as a result of zero scale values."""
374                )
375
376            scale *= scale_change
377            offset += offset_change / scale
378
379            residual = 0.03 * sklearn_metrics.mean_squared_error(
380                current, closest_point, sample_weight=weights
381            )
382
383            weights = np.minimum(
384                1 / np.linalg.norm(current - closest_point, axis=1), 100
385            )
386
387            logger.debug("Scale: %s, Offset: %s, Residual: %s", scale, offset, residual)
388            if abs(residual - previous_residual) < self.error_tolerance:
389                logger.debug("Change in residual below error tolerance, ending loop.}")
390                break
391
392            previous_residual = residual
393
394        return LinearTransformation(scale=scale, offset=offset)

488    def _closest_point_fit(self, no_motion_data: np.ndarray) -> LinearTransformation:
489        """Find linear transformation parameters that minimzes distance to unit sphere.
490
491        This function implements the scipy optimize.minimze function to find the
492        optimial scale and offset parameters that will minimze the error function,
493        which is defined as the distance of the no_motion acceleration data
494        from the unit sphere (where the unit sphere represents the ideal data points
495        under 1g acceleartion due to Earth's gravity). The initial guess for the
496        scale and offset are chosen as 1/0, we provide constrained bounds for the
497        scale and offset parameters to avoid the the case where the scale can be
498        set to 0 and offset to 1/sqrt(3) (exact unit sphere).
499
500        Args:
501            no_motion_data: The acceleration data during periods of no motion.
502
503        Returns:
504            A LinearTransformation object with scale and offset parameters to be applied
505            to acceleration data for calibration.
506
507        Raises:
508            CalibrationError: If the optimization process fails to converge or if the
509                calibration error is not sufficiently minimized.
510        """
511        start_scale = np.ones(3)
512        start_offset = np.zeros(3)
513
514        def get_distance_to_unit_sphere(params: list[float]) -> float:
515            scale = params[:3]
516            offset = params[3:]
517            distances = np.linalg.norm((no_motion_data * scale) + offset, axis=1) - 1
518            return np.sum(distances**2)
519
520        initial_guess = np.concatenate([start_scale, start_offset])
521        data_range = no_motion_data.max() - no_motion_data.min()
522        scale_bounds = [(0.1, None)] * 3
523        offset_bounds = [(-0.5 * data_range, 0.5 * data_range)] * 3
524        bounds = scale_bounds + offset_bounds
525        result = optimize.minimize(
526            get_distance_to_unit_sphere,
527            initial_guess,
528            options={"maxiter": self.max_iterations},
529            bounds=bounds,
530        )
531
532        if result.success:
533            optimal_scale = result.x[:3]
534            optimal_offset = result.x[3:]
535            cal_error_end = np.sqrt(result.fun / len(no_motion_data))
536        else:
537            raise exceptions.CalibrationError("Optimization failed.")
538
539        if cal_error_end >= self.max_calibration_error:
540            cal_error_initial = np.mean(
541                (np.linalg.norm(no_motion_data, axis=1) - 1) ** 2
542            )
543            logger.debug(
544                "Calibration error could not be sufficiently minimized."
545                "Initial Error: %s,  Final Error: %s, Error threshold: %s.",
546                cal_error_initial,
547                cal_error_end,
548                self.max_calibration_error,
549            )
550            raise exceptions.CalibrationError(
551                "Calibration error could not be sufficiently minimized."
552            )
553
554        return LinearTransformation(scale=optimal_scale, offset=optimal_offset)

Calibrates accelerometer data using the default wristpy method.

This is a modification of the method proposed by Van Hees et al. (2014), in which we use all available data, make use of the scipy optimize minimize function to solve for the scaling and offset parameters of a LinearTransformation that will be applied to the data. The minimization function looks to minimize the error between the no_motion_data and the unit sphere.
